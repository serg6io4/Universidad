--Ejercicio 1
SELECT NOMBRE, APELLIDO1, APELLIDO2 FROM DOCENCIA.PROFESORES
WHERE DEPARTAMENTO = 1;

--Ejercicio 2
SELECT NOMBRE, APELLIDO1, APELLIDO2 FROM DOCENCIA.PROFESORES
WHERE DEPARTAMENTO != 3;

--Ejercicio 3
SELECT NOMBRE, APELLIDO1, APELLIDO2, EMAIL"CORREO" FROM DOCENCIA.PROFESORES
WHERE EMAIL LIKE '%lcc.uma.es';

--Ejercicio 4
SELECT NOMBRE, APELLIDO1, APELLIDO2 FROM DOCENCIA.ALUMNOS 
WHERE EMAIL IS NULL;

--Ejercicio 5
SELECT NOMBRE,CURSO, CREDITOS,(TEORICOS/CREDITOS)*100"PORCENTAJE TEÓRICOS",(PRACTICOS/CREDITOS)*100"PORCENTAJE PRÁCTICOS"
FROM DOCENCIA.ASIGNATURAS
WHERE CURSO = 3;

--Ejercicio 6
SELECT ALUMNO,CALIFICACION FROM DOCENCIA.MATRICULAR
WHERE ASIGNATURA = 112
ORDER BY ALUMNO ASC;

--Ejercicio 7
SELECT NOMBRE,(HOMBRES + MUJERES)"POBLACION" FROM DOCENCIA.MUNICIPIO;

--Ejercicio 8
SELECT 'El alumno'||' '||NOMBRE||' '||APELLIDO1||' '||APELLIDO2||' no dispone de correo.'"Correos" FROM DOCENCIA.ALUMNOS 
WHERE EMAIL IS NULL AND GENERO LIKE 'MASC'
UNION
SELECT 'La alumna'||' '||NOMBRE||' '||APELLIDO1||' '||APELLIDO2||' no dispone de correo.'"Correos" FROM DOCENCIA.ALUMNOS
WHERE EMAIL IS NULL AND GENERO LIKE 'FEM';

--Ejercicio 9
SELECT NOMBRE, APELLIDO1, APELLIDO2 FROM DOCENCIA.PROFESORES
WHERE ANTIGUEDAD < TO_DATE('01/01/1990', 'DD/MM/YYYY');

--Ejercicio 10
SELECT NOMBRE, APELLIDO1, APELLIDO2 FROM DOCENCIA.PROFESORES
WHERE (MONTHS_BETWEEN(SYSDATE,FECHA_NACIMIENTO)/12)<30;

--Ejercicio 11
SELECT NOMBRE, APELLIDO1,APELLIDO2, TRUNC((MONTHS_BETWEEN(SYSDATE,ANTIGUEDAD)/12)/3)"TRIENIOS" FROM DOCENCIA.PROFESORES
WHERE TRUNC((MONTHS_BETWEEN(SYSDATE,ANTIGUEDAD)/12)/3)> 3;

--Ejercicio 12
SELECT (REPLACE(NOMBRE,'Bases de Datos','Almacenes de Datos'))"ASIGNATURAS RENOMBRADAS"  FROM DOCENCIA.ASIGNATURAS
WHERE NOMBRE LIKE '%Bases de Datos';

--Ejercicio 13
SELECT NOMBRE,CARACTER, NVL(CREDITOS,NULL) FROM DOCENCIA.ASIGNATURAS
WHERE CARACTER LIKE 'O%';

--Ejercicio 14
SELECT * FROM DOCENCIA.ALUMNOS
WHERE MONTHS_BETWEEN(SYSDATE,FECHA_PRIM_MATRICULA) =2;

--Ejercicio 15
SELECT * FROM DOCENCIA.ALUMNOS
WHERE (MONTHS_BETWEEN(FECHA_NACIMIENTO, FECHA_PRIM_MATRICULA)/12) < 18;

--Ejercicio 16
select *from alumnos
where UPPER(to_char(fecha_prim_matricula,'day')) like '%LUNES%';
